  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  email         text NOT NULL,
  first_name    text,
  tier          text NOT NULL CHECK (tier IN ('beta', 'pro')),
  email_type    text NOT NULL CHECK (
                  email_type IN ('day7_insights', 'day14_feedback', 'day30_trustpilot')
                ),
  scheduled_at  timestamptz NOT NULL,
  sent_at       timestamptz,
  created_at    timestamptz NOT NULL DEFAULT now()
);

CREATE UNIQUE INDEX IF NOT EXISTS email_drip_schedule_email_type_unique
  ON email_drip_schedule(email, email_type);

CREATE INDEX IF NOT EXISTS email_drip_schedule_pending_idx
  ON email_drip_schedule(scheduled_at)
  WHERE sent_at IS NULL;

ALTER TABLE email_drip_schedule ENABLE ROW LEVEL SECURITY;

typescriptimport { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
)

async function scheduleDripEmails(email: string, name: string, tier: 'beta' | 'pro') {
  const firstName = name?.split(' ')[0] ?? null
  const now = new Date()
  const rows = [
    { email, first_name: firstName, tier, email_type: 'day7_insights',
      scheduled_at: new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000).toISOString() },
    { email, first_name: firstName, tier, email_type: 'day14_feedback',
      scheduled_at: new Date(now.getTime() + 14 * 24 * 60 * 60 * 1000).toISOString() },
    { email, first_name: firstName, tier, email_type: 'day30_trustpilot',
      scheduled_at: new Date(now.getTime() + 30 * 24 * 60 * 60 * 1000).toISOString() },
  ]
  const { error } = await supabase
    .from('email_drip_schedule')
    .upsert(rows, { onConflict: 'email,email_type', ignoreDuplicates: true })
  if (error) console.error('[Drip] scheduleDripEmails failed:', error.message)
  else console.log(`[Drip] Scheduled for ${email} (${tier})`)
}
Platzierung: Ruf scheduleDripEmails() direkt nach dem enqueueEmail({ template: "beta-welcome", … }) bzw. enqueueEmail({ template: "pro-welcome", … }) Aufruf auf — nicht nach einem alten direkten resend.emails.send().

STEP 5 — Cron Endpoint
Erstelle app/api/email/drip/cron/route.ts:
typescriptimport { NextRequest, NextResponse } from 'next/server'
import { createClient } from '@supabase/supabase-js'
import { sendDay7InsightsEmail, sendDay14FeedbackEmail, sendDay30TrustpilotEmail } from '@/lib/emails/drip'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
)

export const runtime = 'nodejs'

export async function GET(req: NextRequest) {
  if (req.headers.get('authorization') !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const { data: due, error } = await supabase
    .from('email_drip_schedule')
    .select('id, email, first_name, email_type')
    .is('sent_at', null)
    .lte('scheduled_at', new Date().toISOString())
    .order('scheduled_at', { ascending: true })
    .limit(50)

  if (error) return NextResponse.json({ error: error.message }, { status: 500 })

  const results = { sent: 0, failed: 0 }

  for (const row of due ?? []) {
    try {
      const fn = row.email_type === 'day7_insights' ? sendDay7InsightsEmail
               : row.email_type === 'day14_feedback' ? sendDay14FeedbackEmail
               : row.email_type === 'day30_trustpilot' ? sendDay30TrustpilotEmail
               : null
      if (!fn) continue
      await fn(row.email, row.first_name ?? '')
      await supabase.from('email_drip_schedule').update({ sent_at: new Date().toISOString() }).eq('id', row.id)
      results.sent++
    } catch (e: any) {
      console.error(`[Drip Cron] Failed for ${row.email}:`, e.message)
      results.failed++
    }
  }

  return NextResponse.json({ ok: true, ...results })
}
