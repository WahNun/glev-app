Implementierung — erweitere detectPattern():
typescriptexport function detectPattern(meals: MealForPattern[]): PatternResult {
  // ... bestehende Logik bleibt exakt gleich ...
  const base = existingDetectPattern(meals)  // outcome, sampleSize etc.

  // Kurven-Analyse obendrauf (additiv, bricht nichts)
  const withCurve = meals.filter(m =>
    m.max_bg_180 != null && m.bg_before != null
  )

  if (withCurve.length === 0) {
    return base  // kein Kurvendaten → alles wie vorher
  }

  const hypoCount = withCurve.filter(m => m.had_hypo_window === true).length
  const fastSpikeCount = withCurve.filter(m =>
    m.time_to_peak_min != null && m.time_to_peak_min < 45
  ).length
  const lateDipCount = withCurve.filter(m =>
    m.min_bg_60_180 != null && m.min_bg_60_180 < 80
  ).length

  const peakTimes = withCurve
    .map(m => m.time_to_peak_min)
    .filter((t): t is number => t != null)
  const aucs = withCurve
    .map(m => m.auc_180)
    .filter((a): a is number => a != null)

  return {
    ...base,
    curveDataAvailable: withCurve.length,
    curveInsights: {
      hypoRate: hypoCount / withCurve.length,
      fastSpikeRate: fastSpikeCount / withCurve.length,
      lateDipRate: lateDipCount / withCurve.length,
      avgTimeToPeak: peakTimes.length
        ? peakTimes.reduce((a, b) => a + b, 0) / peakTimes.length
        : null,
      avgAuc: aucs.length
        ? aucs.reduce((a, b) => a + b, 0) / aucs.length
        : null,
    },
  }
}

STEP 5 — suggestAdjustment() um Kurven-Kontext erweitern
In lib/engine/adjustment.ts: wenn curveInsights vorhanden, diese in der Reasoning-Message mitsenden.
typescript// Zusätzliche Messages wenn Kurvendaten vorhanden
if (pattern.curveInsights) {
  const { hypoRate, fastSpikeRate, lateDipRate, avgTimeToPeak } = pattern.curveInsights

  if (hypoRate > 0.2) {
    messages.push({
      key: 'engine.adjustment.hypo_warning',
      params: { rate: Math.round(hypoRate * 100) }
    })
  }

  if (fastSpikeRate > 0.4 && avgTimeToPeak != null) {
    messages.push({
      key: 'engine.adjustment.fast_spike_hint',
      params: { avgMinutes: Math.round(avgTimeToPeak) }
    })
    // Hinweis: Pre-Bolus-Timing, nicht ICR-Änderung
  }

  if (lateDipRate > 0.2) {
    messages.push({
      key: 'engine.adjustment.late_dip_warning',
      params: { rate: Math.round(lateDipRate * 100) }
    })
  }
}
